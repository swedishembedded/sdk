find_package(Zephyr REQUIRED HINTS $ENV{ZEPHYR_BASE})

else()

    app PRIVATE ${CMAKE_CURRENT_LIST_DIR}/../../../../lib/example/example.c)
  unity_generate_test_runner(src/integration.c)
  target_sources(app PRIVATE src/integration.c)
endif()
